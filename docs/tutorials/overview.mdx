Tsim is a quantum circuit sampler that can efficiently sample from Clifford+T circuits with Pauli noise. It is based on ZX-calculus stabilizer rank decomposition and parametrized ZX diagrams, following work of [arXiv:2403.06777](https://arxiv.org/abs/2403.06777).

## Supported Gates

Tsim supports a universal gate set, together with measurement and reset instructions, and Pauli noise channels.

### Clifford Instructions

Tsim supports all instructions [supported by Stim](https://github.com/quantumlib/Stim/wiki/Stim-v1.13-Gate-Reference). Below, we show the standard generating set of Clifford gates:


```python
from tsim import Circuit

c = Circuit("H 0")
print(c.to_matrix())
c.diagram("timeline-svg", height=120)
```

```text title="Output"
[[ 0.70710678+0.j  0.70710678+0.j]
 [ 0.70710678+0.j -0.70710678+0.j]]
```

      var mx = e.clientX - rect.left + wrap.scrollLeft;
      var my = e.clientY - rect.top + wrap.scrollTop;
      var factor = Math.exp(-e.deltaY * 0.01);
      var newScale = Math.min(Math.max(0.02, scale * factor), 40);
      var ratio = newScale / scale;
      scale = newScale;
      apply();
      wrap.scrollLeft = mx * ratio - (e.clientX - rect.left);
      wrap.scrollTop = my * ratio - (e.clientY - rect.top);
    }
  }, { passive: false });
})();

### Non-Clifford Instructions

In addition to Clifford gates, Tsim supports the following non-Clifford gates. Note that all rotation parameters are defined in units of $\pi$.

Computation time and memory requirement scales exponentially with the number of non-Clifford gates.


```python
c = Circuit("T 0")
print(c.to_matrix())
c.diagram("timeline-svg", height=120)
```

```text title="Output"
[[ 1.        +0.j          0.        +0.j        ]
 [-0.        +0.j          0.70710678+0.70710678j]]
```

The `SPP` instruction applies $e^{-i \pi/4 \cdot P}$ (up to global phase) for a Pauli product $P$, which is equivalent to phasing the $-1$ eigenspace of $P$ by $i$. For example, `SPP Z0` is equivalent to the `S` gate, and `SPP X0*X1` is equivalent to `SQRT_XX`. The `SPP_DAG` instruction applies $e^{+i \pi/4 \cdot P}$ (phases by $-i$) instead. The `!` operator inverts the product, swapping `SPP` and `SPP_DAG` behavior.

The second sampling mode is detector sampling. This will sample detector events and observable values. Detector and observable bits can always be obtained by linear transformations of the measurement bits as return by the measurement sampler.
In practice, however, it can be much more efficient to sample detector events directly.


```python
c = Circuit("""
    RX 0
    R 1
    CNOT 0 1
    M 0 1
    DETECTOR rec[-1] rec[-2]
    OBSERVABLE_INCLUDE(0) rec[-1]
""")
sampler = c.compile_detector_sampler()
```

    else if (t == 'Z') return _settings_colors['Zdark'];
    else if (t == 'I') return '#dddddd';
}

function nodeStyle(selected, t) {
    var bold = (t == 7 || t == 8);
    if (selected) return &quot;stroke-width: &quot; + (bold ? &quot;4&quot; : &quot;2&quot;) + &quot;px; stroke: #00f&quot;;
    return &quot;stroke-width: &quot; + (bold ? &quot;3&quot; : &quot;1.5&quot;) + &quot;px&quot;;
}

var symbolGround = {
    draw: function(context, size){
        let s = size/2;

        context.moveTo(0,-s);
        context.lineTo(0,0);

        context.moveTo(-s,0);
        context.lineTo(s,0);

        context.moveTo(-2*s/3,s/3);
        context.lineTo(2*s/3,s/3);

        context.moveTo(-s/3,2*s/3);
        context.lineTo(s/3,2*s/3);
    }
}

function showGraph(tag, graph, width, height, scale, node_size, auto_hbox, show_labels, scalar_str) {
    var ntab = {};

    var groundOffset = 2.5 * node_size;

    graph.nodes.forEach(function(d) {
        ntab[d.name] = d;
        d.selected = false;
        d.previouslySelected = false;
        d.nhd = [];
    });

    var spiders_and_boundaries = graph.nodes.filter(function(d) {
        return d.t != 3;
    });

    graph.links.forEach(function(d) {
        var s = ntab[d.source];
        var t = ntab[d.target];
        d.source = s;
        d.target = t;
        s.nhd.push(t);
        t.nhd.push(s);
    });

    graph.pauli_web.forEach(function(d) {
        var s = ntab[d.source];
        var t = ntab[d.target];
        d.source = s;
        d.target = t;
    });

    var shiftKey;

    // SETUP SVG ITEMS

    var svg = d3.select(tag)
    //.attr(&quot;tabindex&quot;, 1)
        .on(&quot;keydown.brush&quot;, function() {shiftKey = d3.event.shiftKey || d3.event.metaKey;})
        .on(&quot;keyup.brush&quot;, function() {shiftKey = d3.event.shiftKey || d3.event.metaKey;})
    //.each(function() { this.focus(); })
        .append(&quot;svg&quot;)
        .attr(&quot;style&quot;, &quot;max-width: none; max-height: none&quot;)
        .attr(&quot;width&quot;, width)
        .attr(&quot;height&quot;, height);

    var web = svg.append(&quot;g&quot;)
        .attr(&quot;class&quot;, &quot;web&quot;)
        .selectAll(&quot;line&quot;)
        .data(graph.pauli_web)
        .enter().append(&quot;path&quot;)
        .attr(&quot;stroke&quot;, function(d) { return webColor(d.t); })
        .attr(&quot;fill&quot;, &quot;transparent&quot;)
        .attr(&quot;style&quot;, &quot;stroke-width: 7px&quot;);

    var link = svg.append(&quot;g&quot;)
        .attr(&quot;class&quot;, &quot;link&quot;)
        .selectAll(&quot;line&quot;)
        .data(graph.links)
        .enter().append(&quot;path&quot;)
        .attr(&quot;stroke&quot;, function(d) { return edgeColor(d.t); })
        .attr(&quot;fill&quot;, &quot;transparent&quot;)
        .attr(&quot;style&quot;, function(d) { return isBoldEdge(d.t) ? &quot;stroke-width: 3.5px&quot; : &quot;stroke-width: 1.5px&quot;; });

    var brush = svg.append(&quot;g&quot;)
        .attr(&quot;class&quot;, &quot;brush&quot;);

    var node = svg.append(&quot;g&quot;)
        .attr(&quot;class&quot;, &quot;node&quot;)
        .selectAll(&quot;g&quot;)
        .data(graph.nodes)
        .enter().append(&quot;g&quot;)
        .attr(&quot;transform&quot;, function(d) {
            return &quot;translate(&quot; + d.x + &quot;,&quot; + d.y +&quot;)&quot;;
        });

    // Draw a ground symbol connected to the node.
    node.filter(function(d) { return d.ground; })
        .append(&quot;path&quot;)
        .attr(&quot;stroke&quot;, &quot;black&quot;)
        .attr(&quot;style&quot;, &quot;stroke-width: 1.5px&quot;)
        .attr(&quot;fill&quot;, &quot;none&quot;)
        .attr(&quot;d&quot;, &quot;M 0 0 L 0 &quot;+(groundOffset))
        .attr(&quot;class&quot;, &quot;selectable&quot;);
    node.filter(function(d) { return d.ground; })
        .append(&quot;path&quot;)
        .attr(&quot;stroke&quot;, &quot;black&quot;)
        .attr(&quot;style&quot;, &quot;stroke-width: 1.5px&quot;)
        .attr(&quot;fill&quot;, &quot;none&quot;)
        .attr(&quot;d&quot;, d3.symbol().type(symbolGround).size(node_size*1.5))
        .attr(&quot;transform&quot;, &quot;translate(0,&quot;+groundOffset+&quot;)&quot;)
        .attr(&quot;class&quot;, &quot;selectable&quot;);

    node.filter(function(d) { return d.t != 3 &amp;&amp; d.t != 5 &amp;&amp; d.t != 6; })
        .append(&quot;circle&quot;)
        .attr(&quot;r&quot;, function(d) {
            if (d.t == 0) return 0.5 * node_size;
            else if (d.t == 4) return 0.25 * node_size;
            else return node_size;
        })
        .attr(&quot;fill&quot;, function(d) { return nodeColor(d.t); })
        .attr(&quot;stroke&quot;, &quot;black&quot;)
        .attr(&quot;stroke-width&quot;, function(d) { return (d.t == 7 || d.t == 8) ? 3 : 1.5; })
        .attr(&quot;class&quot;, &quot;selectable&quot;);

    var hbox = node.filter(function(d) { return d.t == 3; });

    hbox.append(&quot;rect&quot;)
        .attr(&quot;x&quot;, -0.75 * node_size).attr(&quot;y&quot;, -0.75 * node_size)
        .attr(&quot;width&quot;, node_size * 1.5).attr(&quot;height&quot;, node_size * 1.5)
        .attr(&quot;fill&quot;, function(d) { return nodeColor(d.t); })
        .attr(&quot;stroke&quot;, &quot;black&quot;)
        .attr(&quot;class&quot;, &quot;selectable&quot;);

    // draw a triangle for d.t == 5
    node.filter(function(d) { return d.t == 5; })
        .append(&quot;path&quot;)
        .attr(&quot;d&quot;, &quot;M 0 0 L &quot;+node_size+&quot; &quot;+node_size+&quot; L -&quot;+node_size+&quot; &quot;+node_size+&quot; Z&quot;)
        .attr(&quot;fill&quot;, function(d) { return nodeColor(d.t); })
        .attr(&quot;stroke&quot;, &quot;black&quot;)
        .attr(&quot;class&quot;, &quot;selectable&quot;)
        .attr(&quot;transform&quot;, &quot;translate(&quot; + (-node_size/2) + &quot;, 0) rotate(-90)&quot;);

    // draw a square for Z box: d.t == 6
    node.filter(function(d) { return d.t == 6; })
        .append(&quot;rect&quot;)
        .attr(&quot;x&quot;, -0.75 * node_size).attr(&quot;y&quot;, -0.75 * node_size)
        .attr(&quot;width&quot;, node_size * 1.5).attr(&quot;height&quot;, node_size * 1.5)
        .attr(&quot;fill&quot;, function(d) { return nodeColor(d.t); })
        .attr(&quot;stroke&quot;, &quot;black&quot;)
        .attr(&quot;class&quot;, &quot;selectable&quot;);

    node.filter(function(d) { return d.phase != '' &amp;&amp; d.phase.indexOf('[') === -1; })
        .append(&quot;text&quot;)
        .attr(&quot;y&quot;, 0.7 * node_size + 14)
        .text(function (d) { return d.phase })
        .attr(&quot;text-anchor&quot;, &quot;middle&quot;)
        .attr(&quot;font-size&quot;, &quot;12px&quot;)
        .attr(&quot;font-family&quot;, &quot;monospace&quot;)
        .attr(&quot;fill&quot;, &quot;#00d&quot;)
        .attr('style', 'pointer-events: none; user-select: none;');

    node.filter(function(d) { return d.phase != '' &amp;&amp; d.phase.indexOf('[') !== -1; })
        .append(&quot;text&quot;)
        .attr(&quot;y&quot;, -0.7 * node_size - 8)
        .text(function (d) { return d.phase })
        .attr(&quot;text-anchor&quot;, &quot;middle&quot;)
        .attr(&quot;font-size&quot;, &quot;12px&quot;)
        .attr(&quot;font-family&quot;, &quot;monospace&quot;)
        .attr(&quot;fill&quot;, &quot;black&quot;)
        .attr('style', 'pointer-events: none; user-select: none;');

    if (show_labels) {
        node.append(&quot;text&quot;)
            .attr(&quot;y&quot;, -0.7 * node_size - 8)
            .text(function (d) { return d.name; })
            .attr(&quot;text-anchor&quot;, &quot;middle&quot;)
            .attr(&quot;font-size&quot;, &quot;10px&quot;)
            .attr(&quot;font-family&quot;, &quot;monospace&quot;)
            .attr(&quot;fill&quot;, &quot;#999&quot;)
            .attr('style', 'pointer-events: none; user-select: none;');
    }

    // Display the chosen data fields over the node.
    node.filter(d =&gt; d.vdata.length &gt; 0)
        .append(&quot;text&quot;)
        .attr(&quot;y&quot;, d =&gt; -0.7 * node_size - 14 - 10 * d.vdata.length)
        .attr(&quot;text-anchor&quot;, &quot;middle&quot;)
        .attr(&quot;font-size&quot;, &quot;8px&quot;)
        .attr(&quot;font-family&quot;, &quot;monospace&quot;)
        .attr(&quot;fill&quot;, &quot;#c66&quot;)
        .attr('style', 'pointer-events: none; user-select: none;')
        .selectAll(&quot;tspan&quot;)
        .data(d =&gt; d.vdata)
        .enter()
        .append(&quot;tspan&quot;)
        .attr(&quot;x&quot;, &quot;0&quot;)
        .attr(&quot;dy&quot;, &quot;1.2em&quot;)
        .text(x =&gt; x.join(&quot;: &quot;));

    if (scalar_str != &quot;&quot;) {
        svg.append(&quot;text&quot;)
            .text(scalar_str)
            .attr(&quot;x&quot;, 60).attr(&quot;y&quot;, 40)
            .attr(&quot;text-anchor&quot;, &quot;middle&quot;)
    }

    function update_hboxes() {
        if (auto_hbox) {
            var pos = {};
            hbox.attr(&quot;transform&quot;, function(d) {
                // calculate barycenter of non-hbox neighbours, then nudge a bit
                // to the NE.
                var x=0,y=0,sz=0;
                for (var i = 0; i &lt; d.nhd.length; ++i) {
                    if (d.nhd[i].t != 3) {
                        sz++;
                        x += d.nhd[i].x;
                        y += d.nhd[i].y;
                    }
                }

                offset = 0.25 * scale;

                if (sz != 0) {
                    x = (x/sz) + offset;
                    y = (y/sz) - offset;

                    while (pos[[x,y]]) {
                        x += offset;
                    }
                    d.x = x;
                    d.y = y;
                    pos[[x,y]] = true;
                }

                return &quot;translate(&quot;+d.x+&quot;,&quot;+d.y+&quot;)&quot;;
            });
        }
    }

    update_hboxes();

    var link_curve = function(d) {
        var x1 = d.source.x, x2 = d.target.x, y1 = d.source.y, y2 = d.target.y;
        if (x1 == x2 &amp;&amp; y1 == y2 &amp;&amp; d.num_parallel == 1) {
            var cx1 = x1 - 40;
            var cy1 = y1 - 40;
            var cx2 = x1 + 40;
            var cy2 = y1 - 40;
            return `M ${x1} ${y1} C ${cx1} ${cy1}, ${cx2} ${cy2}, ${x2} ${y2}`;
        } else if (x1 == x2 &amp;&amp; y1 == y2) {
            var pos = d.index + 1;
            var cx1 = x1 - 20 - pos * 10;
            var cy1 = y1 - 20 - pos * 10;
            var cx2 = x1 + 20 + pos * 10;
            var cy2 = y1 - 20 - pos * 10;
            return `M ${x1} ${y1} C ${cx1} ${cy1}, ${cx2} ${cy2}, ${x2} ${y2}`;
        } else if (d.num_parallel == 1) {
            return `M ${x1} ${y1} L ${x2} ${y2}`;
        } else {
            var dx = x2 - x1, dy = y2 - y1;
            var midx = 0.5 * (x1 + x2), midy = 0.5 * (y1 + y2);
            var pos = (d.index / (d.num_parallel-1)) - 0.5;
            var cx = midx - pos * dy;
            var cy = midy + pos * dx;
            return `M ${x1} ${y1} Q ${cx} ${cy}, ${x2} ${y2}`;
            // return `M ${x1} ${y1} L ${x2} ${y2}`;
        }
    };
    link.attr(&quot;d&quot;, link_curve);


    var web_curve = function(d) {
        var x1 = d.source.x, x2 = (x1 + d.target.x)/2, y1 = d.source.y, y2 = (y1 + d.target.y)/2;
        return `M ${x1} ${y1} L ${x2} ${y2}`;
    }
    web.attr(&quot;d&quot;, web_curve);

    // EVENTS FOR DRAGGING AND SELECTION

    node.on(&quot;mousedown&quot;, function(d) {
        if (shiftKey) {
            d3.select(this).selectAll(&quot;.selectable&quot;).attr(&quot;style&quot;, nodeStyle(d.selected = !d.selected, d.t));
            d3.event.stopImmediatePropagation();
        } else if (!d.selected) {
            node.selectAll(&quot;.selectable&quot;).attr(&quot;style&quot;, function(p) { return nodeStyle(p.selected = d === p, p.t); });
        }
    })
        .call(d3.drag().on(&quot;drag&quot;, function(d) {
            var dx = d3.event.dx;
            var dy = d3.event.dy;
            // node.filter(function(d) { return d.selected; })
            //     .attr(&quot;cx&quot;, function(d) { return d.x += dx; })
            //     .attr(&quot;cy&quot;, function(d) { return d.y += dy; });
            node.filter(function(d) { return d.selected; })
                .attr(&quot;transform&quot;, function(d) {
                    d.x += dx;
                    d.y += dy;
                    return &quot;translate(&quot; + d.x + &quot;,&quot; + d.y +&quot;)&quot;;
                });

            update_hboxes();

            link.filter(function(d) { return d.source.selected || d.target.selected ||
                    (auto_hbox &amp;&amp; d.source.t == 3); })
                .attr(&quot;d&quot;, link_curve);
            web.filter(function(d) { return d.source.selected || d.target.selected; })
                .attr(&quot;d&quot;, web_curve);
        }));

    brush.call(d3.brush().keyModifiers(false)
        .extent([[0, 0], [width, height]])
        .on(&quot;start&quot;, function() {
            if (d3.event.sourceEvent.type !== &quot;end&quot;) {
                node.selectAll(&quot;.selectable&quot;).attr(&quot;style&quot;, function(d) {
                    return nodeStyle(
                        d.selected = d.previouslySelected = shiftKey &amp;&amp;
                        d.selected, d.t);
                });
            }
        })
        .on(&quot;brush&quot;, function() {
            if (d3.event.sourceEvent.type !== &quot;end&quot;) {
                var selection = d3.event.selection;
                node.selectAll(&quot;.selectable&quot;).attr(&quot;style&quot;, function(d) {
                    return nodeStyle(d.selected = d.previouslySelected ^
                        (selection != null
                            &amp;&amp; selection[0][0] &lt;= d.x &amp;&amp; d.x &lt; selection[1][0]
                            &amp;&amp; selection[0][1] &lt;= d.y &amp;&amp; d.y &lt; selection[1][1]), d.t);
                });
            }
        })
        .on(&quot;end&quot;, function() {
            if (d3.event.selection != null) {
                d3.select(this).call(d3.event.target.move, null);
            }
        }));
}

function nodeStyle(selected, t) {

var symbolGround = {
    draw: function(context, size){
        let s = size/2;

        context.moveTo(0,-s);
        context.lineTo(0,0);

        context.moveTo(-s,0);
        context.lineTo(s,0);

        context.moveTo(-2*s/3,s/3);
        context.lineTo(2*s/3,s/3);

    function update_hboxes() {
        if (auto_hbox) {
            var pos = {};
            hbox.attr(&quot;transform&quot;, function(d) {

                var x=0,y=0,sz=0;
                for (var i = 0; i &lt; d.nhd.length; ++i) {
                    if (d.nhd[i].t != 3) {
                        sz++;
                        x += d.nhd[i].x;
                        y += d.nhd[i].y;

                offset = 0.25 * scale;

                    y = (y/sz) - offset;

                        x += offset;

                    d.x = x;
                    d.y = y;
                    pos[[x,y]] = true;

                return &quot;translate(&quot;+d.x+&quot;,&quot;+d.y+&quot;)&quot;;
            });
        }

                    d.y += dy;
                    return &quot;translate(&quot; + d.x + &quot;,&quot; + d.y +&quot;)&quot;;
                });

            update_hboxes();

            link.filter(function(d) { return d.source.selected || d.target.selected ||
                    (auto_hbox &amp;&amp; d.source.t == 3); })
                .attr(&quot;d&quot;, link_curve);
            web.filter(function(d) { return d.source.selected || d.target.selected; })
                .attr(&quot;d&quot;, web_curve);
        }));

    brush.call(d3.brush().keyModifiers(false)
        .extent([[0, 0], [width, height]])
        .on(&quot;start&quot;, function() {
            if (d3.event.sourceEvent.type !== &quot;end&quot;) {
                node.selectAll(&quot;.selectable&quot;).attr(&quot;style&quot;, function(d) {
                    return nodeStyle(
                        d.selected = d.previouslySelected = shiftKey &amp;&amp;
                        d.selected, d.t);
                });
            }
        })
        .on(&quot;brush&quot;, function() {
            if (d3.event.sourceEvent.type !== &quot;end&quot;) {
                var selection = d3.event.selection;
                node.selectAll(&quot;.selectable&quot;).attr(&quot;style&quot;, function(d) {
                    return nodeStyle(d.selected = d.previouslySelected ^
                        (selection != null
                            &amp;&amp; selection[0][0] &lt;= d.x &amp;&amp; d.x &lt; selection[1][0]
                            &amp;&amp; selection[0][1] &lt;= d.y &amp;&amp; d.y &lt; selection[1][1]), d.t);
                });
            }
        })
        .on(&quot;end&quot;, function() {
            if (d3.event.selection != null) {
                d3.select(this).call(d3.event.target.move, null);
            }
        }));
}
